📄 ✅ PRACTICAL 3 – STEP BY STEP
🔹 1. CREATE DATABASE
CREATE DATABASE p3;
USE p3;


🔹 2. CREATE TABLES
CREATE TABLE Student (
    StudentID INT PRIMARY KEY,
    Name VARCHAR(50),
    Age INT
);

CREATE TABLE Instructor (
    InstructorID INT PRIMARY KEY,
    Name VARCHAR(50)
);

CREATE TABLE Course (
    CourseID INT PRIMARY KEY,
    CourseName VARCHAR(50),
    Credits INT,
    InstructorID INT,
    FOREIGN KEY (InstructorID) REFERENCES Instructor(InstructorID)
);

CREATE TABLE Enrollment (
    EnrollmentID INT PRIMARY KEY,
    StudentID INT,
    CourseID INT,
    FOREIGN KEY (StudentID) REFERENCES Student(StudentID),
    FOREIGN KEY (CourseID) REFERENCES Course(CourseID)
);

🔥 4. FILTER (WHERE)
👉 Students above age 20
SELECT * FROM Student
WHERE Age > 20;
👉 Filter by instructor
SELECT c.CourseName, i.Name
FROM Course c
JOIN Instructor i ON c.InstructorID = i.InstructorID
WHERE i.Name = 'Dr. Sharma';


🔥 5. SORTING (ORDER BY)
👉 Alphabetical
SELECT * FROM Student
ORDER BY Name;
👉 By credits (descending)
SELECT * FROM Course
ORDER BY Credits DESC;


🔥 6. GROUP BY (Aggregation)
👉 Student count per course
SELECT CourseID, COUNT(*) AS TotalStudents
FROM Enrollment
GROUP BY CourseID;
👉 Average age
SELECT AVG(Age) AS AvgAge
FROM Student;

🔥 7. SUBQUERY
👉 Students above average age
SELECT * FROM Student
WHERE Age > (SELECT AVG(Age) FROM Student);
👉 Courses with more than 1 student
SELECT CourseID
FROM Enrollment
GROUP BY CourseID
HAVING COUNT(*) > 1;

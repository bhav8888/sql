🔥 4. CREATE CURSOR PROCEDURE
DELIMITER //

CREATE PROCEDURE UpdateSalary()
BEGIN
    DECLARE done INT DEFAULT 0;
    DECLARE eid INT;
    DECLARE dept VARCHAR(50);
    DECLARE sal INT;

    DECLARE emp_cursor CURSOR FOR 
        SELECT EmpID, Department, Salary FROM Employees;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    OPEN emp_cursor;

    read_loop: LOOP
        FETCH emp_cursor INTO eid, dept, sal;

        IF done = 1 THEN
            LEAVE read_loop;
        END IF;

        IF dept = 'IT' THEN
            UPDATE Employees
            SET Salary = sal + (sal * 0.10)
            WHERE EmpID = eid;
        END IF;

    END LOOP;

    CLOSE emp_cursor;
END //

DELIMITER ;


🔹 5. CALL PROCEDURE
CALL UpdateSalary();

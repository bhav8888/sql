📄 ✅ PRACTICAL 6 – STEP BY STEP
🔹 1. CREATE DATABASE
CREATE DATABASE p6;
USE p6;


🔹 2. CREATE TABLES
CREATE TABLE Employees (
    EmpID INT PRIMARY KEY,
    Name VARCHAR(50),
    Salary INT
);

CREATE TABLE Employee_Audit (
    AuditID INT AUTO_INCREMENT PRIMARY KEY,
    EmpID INT,
    Action VARCHAR(20),
    ActionDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


🔥 3. TRIGGER FOR INSERT
DELIMITER //

CREATE TRIGGER emp_insert
AFTER INSERT ON Employees
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Audit(EmpID, Action)
    VALUES (NEW.EmpID, 'INSERT');
END //

DELIMITER ;


🔥 4. TRIGGER FOR UPDATE
DELIMITER //

CREATE TRIGGER emp_update
AFTER UPDATE ON Employees
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Audit(EmpID, Action)
    VALUES (NEW.EmpID, 'UPDATE');
END //
DELIMITER ;


🔥 5. TRIGGER FOR DELETE
DELIMITER //

CREATE TRIGGER emp_delete
AFTER DELETE ON Employees
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Audit(EmpID, Action)
    VALUES (OLD.EmpID, 'DELETE');
END //

DELIMITER ;
